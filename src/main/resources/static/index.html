const API_URL = "https://online-sender-2.onrender.com"; async function createClipboard() { const content = document.getElementById('content').value.trim(); const expiry = document.getElementById('expiry').value; if (!content) return showAlert('send-alert', 'Please enter content', false); try { const res = await fetch(`${API_URL}/api/clipboard`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ content, expiryHours: expiry }) }); const data = await res.json(); document.getElementById('generated-code').textContent = data.code; showElement('send-result'); showAlert('send-alert', 'Code generated successfully!', true); } catch { showAlert('send-alert', 'Network error!', false); } } async function retrieveClipboard() { const code = document.getElementById('code-input').value.trim().toUpperCase(); try { const res = await fetch(`${API_URL}/api/clipboard/${code}`); if (!res.ok) throw new Error(); const data = await res.json(); document.getElementById('retrieved-content').textContent = data.content; showElement('receive-result'); showAlert('receive-alert', 'Content loaded!', true); } catch { showAlert('receive-alert', 'Code not found or expired', false); } }
